import os
import sys
import asyncio
import discord
import google.generativeai as genai
from discord.ext import commands
from dotenv import load_dotenv
from datetime import datetime, timezone, timedelta
import re

# ================= LOAD ENV =================
load_dotenv()
DISCORD_TOKEN = os.getenv("DISCORD_TOKEN")
GEMINI_API_KEY = os.getenv("GEMINI_API_KEY")

# ================= GEMINI SETUP =================
genai.configure(api_key=GEMINI_API_KEY)
model = genai.GenerativeModel("gemini-2.5-flash")

# ================= DISCORD SETUP =================
intents = discord.Intents.default()
intents.message_content = True
intents.members = True
bot = commands.Bot(command_prefix="!", intents=intents)

# ================= CONFIG =================
LOG_CHANNEL_ID = 1475415105830129706
JUDGEMENT_LOGO = "https://i.imgur.com/8Km9tLL.png"

STRICT_WORDS = [
    "madarchod", "bhosdike", "chutiya",
    "lauda", "randi", "mc", "bc"
]

LINK_REGEX = re.compile(r"https?://|discord\.gg|www\.", re.IGNORECASE)

# ================= AI STRICT CHECK =================
async def check_abuse_strict(message_content):
    try:
        prompt = f"""
You are an EXTREMELY STRICT Discord moderation AI.

Rules:
- Any insult = ABUSIVE
- Any sexual slur = ABUSIVE
- Any harassment = ABUSIVE
- Even light insult = ABUSIVE
- Be ultra strict.

Reply ONLY with:
ABUSIVE
or
SAFE

Message:
{message_content}
"""
        response = model.generate_content(prompt)
        result = response.text.strip().upper()
        return "ABUSIVE" in result
    except Exception as e:
        print("AI Error:", e)
        return False

# ================= PUNISH FUNCTION =================
async def punish_user(message, reason, minutes):
    try:
        await message.delete()
        timeout_duration = timedelta(minutes=minutes)
        await message.author.timeout(timeout_duration, reason=reason)

        log_channel = bot.get_channel(LOG_CHANNEL_ID)
        if log_channel is None:
            print("Log channel not found")
            return

        embed = discord.Embed(
            title="⚖️ Judgement Taken",
            color=discord.Color.from_rgb(255, 50, 50),
            timestamp=datetime.now(timezone.utc)
        )

        embed.set_thumbnail(url=message.author.display_avatar.url)
        embed.description = (
            f"**Reason in spoiler**\n"
            f"||{reason}||\n\n"
            f"Muted for **{minutes} minutes**\n\n"
            f"**Judgement Taken By Ender Security**"
        )

        embed.set_image(url=JUDGEMENT_LOGO)
        await log_channel.send(embed=embed)
    except Exception as e:
        print("Punish Error:", e)

# ================= EVENTS =================
@bot.event
async def on_ready():
    print(f"✅ Logged in as {bot.user}")

@bot.event
async def on_message(message):
    if message.author.bot:
        return

    content = message.content.lower()

    # -------- STRICT WORD CHECK (20 MIN) --------
    for word in STRICT_WORDS:
        if word in content:
            await punish_user(message, "STRICT WORD VIOLATION", 20)
            return

    # -------- LINK CHECK (10 MIN) --------
    if LINK_REGEX.search(content):
        await punish_user(message, "UNAUTHORIZED LINK SHARING", 10)
        return

    # -------- AI INSULT CHECK (15 MIN) --------
    is_abusive = await check_abuse_strict(message.content)
    if is_abusive:
        await punish_user(message, "AI STRICT ANTI INSULT DETECTION", 15)
        return

    await bot.process_commands(message)

# ================= RESTART COMMAND =================
@bot.command(name="enderarmy")
@commands.has_permissions(administrator=True)
async def enderarmy(ctx, action: str):
    if action.lower() == "restart":
        await ctx.send("♻️ Bot is restarting... Please wait...")

        # Optional: log restart
        log_channel = bot.get_channel(LOG_CHANNEL_ID)
        if log_channel:
            await log_channel.send("⚠️ Bot was restarted by an administrator.")

        # Close bot and exit Python process
        await bot.close()
        sys.exit()
    else:
        await ctx.send("❌ Unknown action. Use `.enderarmy restart`.")

# ================= ERROR HANDLING =================
@enderarmy.error
async def enderarmy_error(ctx, error):
    if isinstance(error, commands.MissingPermissions):
        await ctx.send("❌ You must be an administrator to use this command.")

# ================= RUN BOT =================
bot.run(DISCORD_TOKEN)
